## 更新记录

### 2026-06-11：前端 API 规范修正

**修改文件**：`React/src/utils/request.js`

**改动 1：Token 传递方式改为标准 Header**
- **原设计**：Token 放在请求体 `body.authorization` 中（Mock.js 妥协方案）
- **新设计**：Token 放在 `Authorization: Bearer <token>` Header 中（RFC 7235 标准）
- **原因**：Mock.js 即将废弃，后端采用标准 RESTful 设计

**改动 2：响应拦截器改用 HTTP 状态码**
- **原设计**：判断 `response.data.code !== 200`
- **新设计**：判断 `response.status >= 200 && response.status < 300`
- **原因**：HTTP 状态码应由响应头表示，业务状态码仅表示更细粒度的错误类型

**影响范围**：
- ✅ 前端 `request.js` 已修改
- ⚠️ 后端 `middlewares/auth.js` 需对应修改（从 Header 读取 Token）
- ⚠️ 后端响应格式需统一（使用 HTTP 状态码 + 业务 code）

**决策**：采用方案 A（修正前端 API 以符合 REST 规范）

#### 3.1.1 注册接口（`POST /api/auth/register`）

**前端请求**：
```javascript
POST /api/register
Content-Type: application/json

{
  "username": "alice",
  "password": "123456"
}
```

**后端实现**（`controllers/authController.js`）：
```javascript
const db = require('../config/db');
const bcrypt = require('bcrypt');
const jwt = require('jsonwebtoken');

const register = async (req, res) => {
  try {
    const { username, password } = req.body;

    // 1. 检查用户名是否已存在
    const [existing] = await db.query(
      'SELECT uid FROM users WHERE username = ?',
      [username]
    );
    if (existing.length > 0) {
      return res.status(400).json({
        code: 400,
        message: '用户名已存在',
        data: null
      });
    }

    // 2. 密码加密
    const hashedPassword = await bcrypt.hash(password, 10);

    // 3. 插入数据库
    const [result] = await db.query(
      'INSERT INTO users (username, password) VALUES (?, ?)',
      [username, hashedPassword]
    );

    // 4. 签发 JWT
    const token = jwt.sign(
      { uid: result.insertId },
      process.env.JWT_SECRET,
      { expiresIn: '10h' }
    );

    // 5. 返回响应
    res.status(201).json({
      code: 200,
      message: '注册成功',
      data: {
        token: token,
        uid: result.insertId
      }
    });
  } catch (err) {
    next(err);  // 交给错误处理中间件
  }
};

module.exports = { register };
```

**前端请求**：

```javascript
const login = async (req, res, next) => {
  try {
    const { username, password } = req.body;

    // 1. 查询用户
    const [users] = await db.query(
      'SELECT uid, username, password FROM users WHERE username = ?',
      [username]
    );
    if (users.length === 0) {
      return res.status(401).json({
        code: 401,
        message: '用户名或密码错误',
        data: null
      });
    }

    const user = users[0];

    // 2. 验证密码
    const match = await bcrypt.compare(password, user.password);
    if (!match) {
      return res.status(401).json({
        code: 401,
        message: '用户名或密码错误',
        data: null
      });
    }

    // 3. 签发 JWT
    const token = jwt.sign(
      { uid: user.uid },
      process.env.JWT_SECRET,
      { expiresIn: '10h' }
    );

    // 4. 返回响应
    res.json({
      code: 200,
      message: '登录成功',
      data: {
        token: token,
        uid: user.uid
      }
    });
  } catch (err) {
    next(err);
  }
};

module.exports = { login, register };
```

#### 3.1.3 Token 验证中间件（`middlewares/auth.js`）

**标准做法**：从 `Authorization` Header 读取 Token（Bearer 方案）
```javascript
const jwt = require('jsonwebtoken');

const authMiddleware = (req, res, next) => {
  // 标准做法：从 Authorization Header 读取 token
  // 前端已修改为：config.headers.Authorization = `Bearer ${token}`
  const authHeader = req.headers.authorization;

  if (!authHeader) {
    return res.status(401).json({
      code: 401,
      message: '请先登录',
      data: null
    });
  }

  // 去掉 "Bearer " 前缀
  const token = authHeader.split(' ')[1];

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;  // 附加到 req 对象，后续路由可访问
    next();
  } catch (err) {
    return res.status(401).json({
      code: 401,
      message: 'Token 无效或已过期',
      data: null
    });
  }
};

module.exports = authMiddleware;
```

#### 3.2.1 获取视频列表（`GET /api/video`）

GET /api/video?sort=-date&page=1&element=16&q=关键词

**后端实现**（`controllers/videoController.js`）：
```javascript
const getVideoList = async (req, res, next) => {
  try {
    const { sort, page = 1, element = 16, q } = req.query;

    // 1. 构建 SQL 查询
    let sql = 'SELECT * FROM videos WHERE 1=1';
    let params = [];

    // 关键词搜索
    if (q) {
      sql += ' AND (title LIKE ? OR info LIKE ?)';
      const kw = `%${q}%`;
      params.push(kw, kw);
    }

    // 排序
    if (sort) {
      const order = sort.startsWith('-') ? 'DESC' : 'ASC';
      const field = sort.replace(/^[+-]/, '');
      sql += ` ORDER BY ${field} ${order}`;
    }

    // 分页
    const offset = (page - 1) * element;
    sql += ' LIMIT ? OFFSET ?';
    params.push(Number(element), Number(offset));

    // 2. 执行查询
    const [videos] = await db.query(sql, params);

    // 3. 查询总数（用于分页）
    const [totalResult] = await db.query(
      'SELECT COUNT(*) as total FROM videos'
    );
    const total = totalResult[0].total;

    // 4. 返回响应
    res.json({
      code: 200,
      message: 'ok',
      data: videos,
      total: total
    });
  } catch (err) {
    next(err);
  }
};

module.exports = { getVideoList };
```

#### 3.2.2 获取视频详情（`GET /api/video/:vid`）

**后端实现**：

const getVideo = async (req, res, next) => {
  try {
    const { vid } = req.params;

    // 1. 查询视频信息
    const [videos] = await db.query(
      `SELECT v.*, u.username, u.profilePictureUrl
       FROM videos v
       JOIN users u ON v.uploader_uid = u.uid
       WHERE v.vid = ?`,
      [vid]
    );

    if (videos.length === 0) {
      return res.status(404).json({
        code: 404,
        message: '视频不存在',
        data: null
      });
    }

    const video = videos[0];

    // 2. 查询标签列表
    const [tags] = await db.query(
      `SELECT t.* FROM tags t
       JOIN video_tags vt ON t.tid = vt.tid
       WHERE vt.vid = ?`,
      [vid]
    );

    video.tagList = tags;

    // 3. 如果已登录，添加到历史记录
    if (req.user) {
      await db.query(
        `INSERT IGNORE INTO user_history (uid, type, item_id)
         VALUES (?, 'video', ?)`,
        [req.user.uid, vid]
      );
    }

    // 4. 返回响应
    res.json({
      code: 200,
      message: 'ok',
      data: video
    });
  } catch (err) {
    next(err);
  }
};

module.exports = { getVideo, getVideoList };
```

    await db.query(
      'INSERT INTO user_likes (uid, type, item_id) VALUES (?, "video", ?)',
      [uid, vid]
    );

    // 3. 更新视频的 likeCount
    await db.query(
      'UPDATE videos SET likeCount = likeCount + 1 WHERE vid = ?',
      [vid]
    );

    // 4. 返回响应
    res.json({
      code: 200,
      message: '点赞成功',
      data: null
    });
  } catch (err) {
    next(err);
  }
};

module.exports = { likeVideo };
```
